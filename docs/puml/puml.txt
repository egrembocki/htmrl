title US-01 Input Data Ingestion (Current InputHandler)

actor Developer
participant "InputHandler" as IH
participant "Filesystem" as FS
participant "File Parser" as Parser

Developer -> IH: input_data(input_source, required_columns?)
activate IH

alt input_source is str path
  IH -> FS: exists(path)?
  FS --> IH: true/false

  alt file exists
    IH -> Parser: _load_from_file(path)
    activate Parser
    Parser --> IH: raw_data
    deactivate Parser
    IH -> IH: _raw_to_sequence(raw_data)
  else extension indicates file but missing
    IH --> Developer: FileNotFoundError
    deactivate IH
    return
  end
else input_source is in-memory payload
  IH -> IH: _raw_to_sequence(input_source)
end

opt required_columns provided
  IH -> IH: _apply_required_columns(required_columns)
end

IH -> IH: _validate_data(required_columns)
IH --> Developer: normalized records

deactivate IH
@enduml

title US-02 Validate Data (Current InputHandler._validate_data)

actor Developer
participant "InputHandler" as IH
participant "Logger" as LOG

Developer -> IH: _validate_data(required_columns?)
activate IH

alt self._data is not list[dict] or empty
  IH -> LOG: error/warning(...)
  IH --> Developer: ValueError
else structure is valid
  IH -> IH: inspect columns for duplicates/all-None

  alt invalid columns detected
    IH -> LOG: warning("invalid dataset structure")
    IH --> Developer: ValueError
  else column checks pass
    opt required_columns provided
      IH -> IH: verify required columns are present
      alt missing required columns
        IH -> LOG: warning("missing required columns")
        IH --> Developer: ValueError
      end
    end

    IH -> LOG: info("data has been validated")
    IH --> Developer: True
  end
end

deactivate IH
@enduml
