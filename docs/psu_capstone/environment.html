 89    def _to_serializable(self, value: Any) -> Any:
 90        """Convert numpy values to JSON-friendly Python types.
 91
 92        Call chain for runtime observation/action values:
 93            reset_bridge/step_bridge -> observation_to_inputs ->
 94            _space_value_to_inputs -> _flatten_value -> _to_serializable
 95
 96        Call chain for space metadata values:
 97            get_observation_spec/get_action_spec -> _space_to_schema ->
 98            _to_serializable
 99        """
100
101        # Local role: normalize data types so downstream code can treat values
102        # as plain Python for flattening, logging, and metadata output.
103
104        if isinstance(value, np.ndarray):
105            # Arrays are not JSON-native; convert to nested Python lists.
106            return value.tolist()
107        if isinstance(value, np.generic):
108            # NumPy scalar wrappers (np.float32, np.int64, etc.) -> Python scalar.
109            return value.item()
110        if isinstance(value, tuple):
111            # Normalize tuples recursively so mixed nested outputs stay consistent.
112            return [self._to_serializable(item) for item in value]
113        if isinstance(value, list):
114            # Normalize each list element recursively.
115            return [self._to_serializable(item) for item in value]
116        if isinstance(value, dict):
117            # Normalize nested mappings used by Dict observation/action spaces.
118            return {key: self._to_serializable(item) for key, item in value.items()}
119
120        # Primitive Python values are already serializable.
121        return value

123    def _space_to_schema(self, space: gym.Space[Any]) -> dict[str, Any]:
124        """Take a Gym space object and convert it into a plain, serializable description of that space.
125
126        This does not return runtime values. It only describes shape/type
127        details for debugging, introspection, or setup checks.
128        """
129
130        # Semantic role: describe the structure/constraints of a space as a schema.
131        # Adapter-facing role: expose Gym space details in one predictable,
132        # serializable dictionary structure for outside inspection.
133
134        if isinstance(space, gym.spaces.Discrete):
135            return {
136                "space_type": "Discrete",
137                "n": space.n,
138                "values": list(range(space.n)),
139            }
140
141        if isinstance(space, gym.spaces.MultiDiscrete):
142            nvec = space.nvec.tolist()
143            return {
144                "space_type": "MultiDiscrete",
145                "nvec": nvec,
146                "values": [list(range(int(n))) for n in nvec],
147            }
148
149        if isinstance(space, gym.spaces.MultiBinary):
150            return {
151                "space_type": "MultiBinary",
152                "n": self._to_serializable(space.n),
153                "values": [0, 1],
154            }
155
156        if isinstance(space, gym.spaces.Box):
157            return {
158                "space_type": "Box",
159                "shape": list(space.shape),
160                "dtype": str(space.dtype),
161                "low": self._to_serializable(space.low),
162                "high": self._to_serializable(space.high),
163            }
164
165        if isinstance(space, gym.spaces.Tuple):
166            return {
167                "space_type": "Tuple",
168                "spaces": [self._space_to_schema(subspace) for subspace in space.spaces],
169            }
170
171        if isinstance(space, gym.spaces.Dict):
172            return {
173                "space_type": "Dict",
174                "spaces": {
175                    key: self._space_to_schema(subspace) for key, subspace in space.spaces.items()
176                },
177            }
178
179        return {
180            "space_type": type(space).__name__,
181            "sample": self._to_serializable(space.sample()),
182        }

184    def _flatten_value(self, value: Any, prefix: str) -> dict[str, Any]:
185        """Flatten nested values into a simple key/value map.
186
187        Args:
188            value: The value to flatten (dict, list, or scalar).
189            prefix: Key prefix used to build namespaced output keys.
190
191        Returns:
192            A flat dictionary mapping prefixed string keys to scalar values.
193
194        Example:
195            a nested value can turn into keys like observation_0 or
196            action_price_open.
197        """
198
199        # Normalize first so flattening only handles plain Python containers.
200        serializable = self._to_serializable(value)
201
202        # Local role: recursively turn nested values into flat key/value pairs
203        # the Brain can consume as a simple input mapped dictionary.
204        if isinstance(serializable, dict):
205            flattened: dict[str, Any] = {}
206            for key, item in serializable.items():
207                flattened.update(self._flatten_value(item, f"{prefix}_{key}"))
208            return flattened
209
210        if isinstance(serializable, list):
211            flattened = {}
212            for index, item in enumerate(serializable):
213                flattened.update(self._flatten_value(item, f"{prefix}_{index}"))
214            return flattened
215
216        return {prefix: serializable}

class FinGym(typing.Generic[~ObsType, ~ActType]): 
 20class FinGym(gym.Env):
 21    """Gymnasium environment for row-wise tabular/financial datasets.
 22
 23    This environment treats a table like a time-indexed sequence where each
 24    row is one timestep. The observation is a numeric vector built from selected
 25    feature columns. Rewards can optionally be derived from the delta of a
 26    target column between consecutive rows.
 27
 28    Actions are 0=hold, 1=long, 2=short. If ``target_column`` is not provided,
 29    rewards are always ``0.0``. Non-numeric feature values are coerced to numeric
 30    and missing values are filled with ``0.0`` for stable observation vectors.
 31
 32    Args:
 33        data_source: Table source as DataFrame, file path, or mapping of
 34            columns to value sequences.
 35        feature_columns: Optional ordered list of columns used as observation
 36            features. If omitted, all columns except ``target_column`` are used.
 37        target_column: Optional numeric column used to compute reward from
 38            row-to-row deltas.
 39        max_rows: Optional cap on number of rows read from the dataset.
 40
 41    Attributes:
 42        metadata: Gymnasium render mode metadata.
 43
 44    Raises:
 45        ValueError: If ``max_rows`` is not greater than 1, if fewer than 2 rows are
 46            available after loading, or if ``target_column`` is not found in the data.
 47    """
 48
 49    metadata = {"render_modes": ["human"]}
 50
 51    def __init__(
 52        self,
 53        data_source: pd.DataFrame | str | Path | Mapping[str, Sequence[Any]],
 54        feature_columns: list[str] | None = None,
 55        target_column: str | None = None,
 56        max_rows: int | None = None,
 57    ) -> None:
 58        super().__init__()
 59
 60        # Load the raw data and optionally cap it to max_rows for testing/prototyping
 61        self._raw_frame = self._load_frame(data_source)
 62
 63        if max_rows is not None:
 64            if max_rows <= 1:
 65                raise ValueError("max_rows must be greater than 1.")
 66            self._raw_frame = self._raw_frame.iloc[:max_rows].copy()
 67
 68        if len(self._raw_frame) < 2:
 69            raise ValueError("FinGym requires at least 2 rows to step through data.")
 70
 71        self._target_column = target_column
 72        self._feature_columns = self._resolve_feature_columns(feature_columns, target_column)
 73
 74        # Normalize all selected feature columns into a dense float matrix that
 75        # can be consumed directly by Gym/agents. We convert to numeric and converting
 76        # non-numeric values to NaN, then fill NaNs with 0.0 for stability, and
 77        # cast to float32 for memory efficiency and neural network compatibility.
 78        features = self._raw_frame[self._feature_columns].apply(pd.to_numeric, errors="coerce")
 79        features = features.fillna(0.0).astype(np.float32)
 80        self._feature_matrix = features.to_numpy(dtype=np.float32)
 81
 82        # If a target column was specified, normalize it to float32 for reward calculations.
 83        # Target values represent the signal we want the agent to learn to predict.
 84        self._target_values: np.ndarray | None = None
 85
 86        if self._target_column is not None:
 87            if self._target_column not in self._raw_frame.columns:
 88                raise ValueError(f"Unknown target column: {self._target_column}")
 89            target_series = pd.to_numeric(

192    def _reward_from_action(self, action: int) -> float:
193        """Compute reward from target delta for hold/long/short actions.
194
195        This method realizes the core financial logic: agents earn reward when
196        their action aligns with the next price movement.
197
198        Reward is based on ``target[t+1] - target[t]``:
199            - long (1): positive delta is rewarded, negative delta penalized
200            - short (2): negative delta is rewarded, positive delta penalized
201            - hold (0): no action, no reward (delta is ignored)
202
203        Returns 0.0 if target column is not defined or we're at the final row.
204        """
205
206        # Cannot compute reward if no target or at the final row (no next value to compare)
207        if self._target_values is None or self._current_row >= self._row_count - 1:
208            return 0.0
209
210        # Calculate the price movement from current to next timestep
211        current_value = self._target_values[self._current_row]
212        next_value = self._target_values[self._current_row + 1]
213        delta = float(next_value - current_value)
214
215        # Map action to reward: long benefits from positive delta, short from negative
216        if action == 1:  # long
217            return delta
218        if action == 2:  # short
219            return -delta
220
221        return 0.0  # hold: no reward

250    def step(self, action: int) -> tuple[np.ndarray, float, bool, bool, dict[str, Any]]:
251        """Advance one row and return standard Gymnasium step tuple.
252
253        Each step represents processing one row of the dataset. The environment
254        terminates after the final valid transition (row_count-2 -> row_count-1),
255        at which point there is no next row to transition to.
256
257        Args:
258            action: One of 0 (hold), 1 (long), or 2 (short)
259
260        Returns:
261            (observation, reward, terminated, truncated, info)
262
263        Raises:
264            ValueError: If ``action`` is not a valid value in the action space.
265        """
266
267        # Validate action is in the discrete space {0, 1, 2}
268        if not self.action_space.contains(action):
269            raise ValueError(f"Invalid action: {action}")
270
271        # Compute reward based on how well the action aligns with the next price move
272        reward = self._reward_from_action(action)
273
274        # Episode ends when we reach the last row (no next row to transition to for reward)
275        terminated = self._current_row >= self._row_count - 2
276        truncated = False  # We don't use Gymnasium truncation; only proper termination
277
278        # Only advance if not terminated; at termination we stay at the final valid row
279        if not terminated:
280            self._current_row += 1
281
282        # Fetch the observation (feature vector) for the current row
283        observation = self._current_observation()
284        # Include metadata for debugging and post-episode analysis
285        info = {
286            "row_index": self._current_row,
287            "reward_source": self._target_column,
288        }
289
290        return observation, reward, terminated, truncated, info

This environment treats a table like a time-indexed sequence where each row is one timestep. The observation is a numeric vector built from selected feature columns. Rewards can optionally be derived from the delta of a target column between consecutive rows.

Actions are 0=hold, 1=long, 2=short. If target_column is not provided, rewards are always 0.0. Non-numeric feature values are coerced to numeric and missing values are filled with 0.0 for stable observation vectors.

Args: data_source: Table source as DataFrame, file path, or mapping of columns to value sequences. feature_columns: Optional ordered list of columns used as observation features.
