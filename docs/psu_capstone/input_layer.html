184    def _load_file_to_dataframe(self, input_source: str | os.PathLike[str]) -> pd.DataFrame:
185        """Load data from a file path into a pandas DataFrame based on the file extension.
186
187        Args:
188            input_source: A string or os.PathLike object representing the file path to load.
189
190        Returns:
191            A pandas DataFrame containing the data loaded from the specified file.
192
193        Raises:
194            ValueError: If the file extension is not supported for loading.
195            FileNotFoundError: If the specified file does not exist.
196
197        """
198
199        path = Path(input_source)
200        if not path.exists() or not path.is_file():
201            raise FileNotFoundError(f"File not found: {path}")
202        ext = path.suffix.lower()
203        self.logger.info("Loading data from %s", path)
204
205        if ext == ".csv":
206            # Try to read CSV and detect metadata rows
207            df = pd.read_csv(path)
208
209            # Check if first row contains type information (metadata)
210            if len(df) > 0 and all(
211                isinstance(val, str) and val in ["datetime", "float", "int", "str", "T", ""]
212                for val in df.iloc[0].values
213                if pd.notna(val)
214            ):
215                # Skip metadata rows and re-read
216                self.logger.info("Detected metadata rows in CSV, skipping them")
217                df = pd.read_csv(path, skiprows=[1, 2])
218
219            return df
220        if ext == ".json":
221            return pd.read_json(path)
222        if ext == ".xlsx":
223            return pd.read_excel(
224                path
225            )  # use openpyxl engine for XLSX files, which is the default in recent pandas versions and supports modern Excel formats. For legacy XLS files, xlrd may be required but is not guaranteed to be available in all environments.
226        if ext == ".xls":
227            return pd.read_excel(
228                path
229            )  # use xlrd engine for legacy XLS files if available, otherwise fallback to openpyxl which may not support XLS
230        if ext == ".parquet":
231            return pd.read_parquet(path)
232        if ext == ".txt":
233            return pd.DataFrame(
234                {"value": path.read_text(encoding="utf-8").splitlines(keepends=True)}
235            )  # build inline DataFrame for text files, treating each line as a separate record
236        raise ValueError(f"Unsupported file type: {ext}")

238    def _coerce_non_file_to_dataframe(self, input_source: Any) -> pd.DataFrame:
239        """Convert in-memory data structures into a pandas DataFrame for normalization and validation.
240
241        Args:
242            input_source: The raw input data, which can be a variety of in-memory types
243
244        Returns:
245            A pandas DataFrame representing the input data, ready for processing and validation.
246
247        Raises:
248            TypeError: If the input data type is not supported for conversion to a DataFrame.
249
250        """
251
252        if isinstance(input_source, bytes):
253            input_source = bytearray(input_source)
254
255        if isinstance(input_source, pd.DataFrame):
256            return input_source.copy()
257
258        elif isinstance(input_source, pd.Series):
259            return input_source.to_frame()
260
261        elif isinstance(input_source, np.ndarray):
262            self._contains_multidimensional_data = input_source.ndim > 1
263            return pd.DataFrame(input_source)
264
265        elif isinstance(input_source, Mapping):
266            return pd.DataFrame(input_source)
267
268        elif isinstance(input_source, bytearray):
269            if b"," in input_source or b"\n" in input_source:
270                try:
271                    text_payload = input_source.decode("utf-8")
272                except UnicodeDecodeError:
273                    text_payload = input_source.decode("latin-1")
274                if all(ch.isprintable() or ch.isspace() for ch in text_payload):
275                    try:
276                        csv_df = pd.read_csv(StringIO(text_payload), dtype=str)
277                        if not csv_df.empty and csv_df.columns.size > 0:
278                            return csv_df
279                    except Exception:
280                        pass
281            return pd.DataFrame({"value": list(input_source)})
282
283        elif isinstance(input_source, Sequence) and not isinstance(
284            input_source, (str, bytes, bytearray)
285        ):
286            # Handle sequences of mappings (e.g., list of dicts) and sequences of sequences (e.g., list of lists)
287            if not input_source:
288                return pd.DataFrame()
289            first = input_source[0]
290            if isinstance(first, Mapping):
291                return pd.DataFrame(list(input_source))
292            if isinstance(first, Sequence) and not isinstance(first, (str, bytes, bytearray)):
293                self._contains_multidimensional_data = True
294                return pd.DataFrame([list(row) for row in input_source])
295            return pd.DataFrame({"value": list(input_source)})
296
297        elif isinstance(input_source, (str, int, float)):
298            return pd.DataFrame({"value": [input_source]})
299
300        else:
301
302            raise TypeError(f"Unsupported data type for conversion: {type(input_source)}")

304    def _process_dataframe(
305        self, df: pd.DataFrame, required_columns: list[str] | None = None
306    ) -> pd.DataFrame:
307        """Normalize datetime columns, enforce required columns, detect repeating temporal patterns, and handle missing values.
308
309        Args:
310            df: The input DataFrame to process and normalize.
311            required_columns: A list of column names that must be present in the output DataFrame.
312
313        Returns:
314            A processed and normalized DataFrame that meets the specified requirements.
315        """
316        if df.empty:
317            self._repeating_columns = []
318            return df
319
320        if not df.columns.is_unique:
321            duplicates = df.columns[df.columns.duplicated()].tolist()
322            logger.info("Duplicate columns detected: %s", duplicates)
323
324            df = df.loc[:, ~df.columns.duplicated()]
325            logger.info("Removed duplicate columns, remaining columns: %s", df.columns.tolist())
326
327        if required_columns is not None:
328            missing = [col for col in required_columns if col not in df.columns]
329            for col in missing:
330                df[col] = None
331            df = df[required_columns]
332            logger.info("Added missing required columns: %s", missing)
333
334        # check for existing datetime
335        df = self._normalize_datetime_columns(df)
336
337        # check for consistent primitive types and coerce numeric types for downstream processing
338        df = self._normalize_column_types(df)
339
340        # fill missing values in numeric columns with the mean of the column
341        df = self._fill_missing_values(df)
342
343        self._isRepeating, self._repeating_columns = self._detect_repeating_values(df, threshold=3)
344
345        return df

347    def _normalize_datetime_columns(self, df: pd.DataFrame) -> pd.DataFrame:
348        """Detect and normalize datetime columns while preserving datetimelike objects."""
349
350        dataframe = df.copy()
351
352        for col in dataframe.columns:
353            series = dataframe[col]
354            normal = series.dropna()
355            if normal.empty:
356                continue
357
358            sample = normal.iloc[0]
359            parsed: pd.Series | None = None
360
361            if isinstance(sample, (datetime, pd.Timestamp, time.struct_time, np.datetime64)):
362
363                continue  # already a datetimelike type, no parsing needed
364
365            elif isinstance(sample, str):
366                maybe_time = any(token in sample for token in ("-", ":", "T", "/", " "))
367                if maybe_time:
368                    parsed = pd.to_datetime(series)
369
370            if parsed is not None and parsed.notna().sum() == normal.shape[0]:
371                dataframe[col] = parsed
372
373        dataframe = dataframe.dropna()
374
375        return dataframe

377    def _normalize_column_types(self, df: pd.DataFrame) -> pd.DataFrame:
378        """Enforce consistent primitive types across columns and coerce numeric types for downstream processing.
379
380        Args:
381            df: The input DataFrame to analyze and normalize column types.
382
383        Returns:
384            A DataFrame with consistent primitive types across columns, where numeric columns are coerced to appropriate numeric types for downstream processing, while non-numeric columns are preserved as-is.
385
386        Raises:
387            ValueError: If the DataFrame contains unsupported types or mixed primitive types within a column that cannot be normalized.
388
389
390        """
391
392        dataframe = df.copy()
393
394        for col in dataframe.columns:
395
396            normal = dataframe[col].dropna()
397
398            logger.info(
399                "Analyzing column %-24s dtype=%-12s non-null=%5d",
400                col,
401                str(normal.dtype),
402                normal.shape[0],
403            )
404
405            # store a set of types present in the column to detect unsupported or mixed types
406            primitive_types = {type(value) for value in normal}
407
408            # check for any non-primitive types that would indicate corrupt or unsupported data
409            # against the inline set of allowed primitive types (str, bool, int, float, and their numpy equivalents)
410            has_non_primitive = any(
411                t
412                not in {
413                    str,
414                    bool,
415                    int,
416                    float,
417                    datetime,
418                    pd.Timestamp,
419                    np.int64,
420                    np.float64,
421                    np.int32,
422                    np.float32,
423                    np.datetime64,
424                    np.ndarray,
425                }
426                for t in primitive_types
427            )
428
429            if normal.empty:  # go to next column if all values are None/NaN
430                continue
431
432            elif has_non_primitive:  # look for unsupported types in a mapping {key: type(value)}
433                logger.error(
434                    "Unsupported types detected in column '%s': %s",
435                    col,
436                    sorted(t.__name__ for t in primitive_types),
437                )
438                raise ValueError(f"Corrupt data detected in column '{col}': unsupported value type")
439
440            # enforce consistent primitive types across the columns, all values are the same primitive type
441            # check for more than one type in the col
442            elif len(primitive_types) > 1:
443                logger.warning(
444                    "Column '%s' has multiple types: %s",
445                    col,
446                    sorted(t.__name__ for t in primitive_types),
447                )
448
449                # attempt to convert to numeric
450                if primitive_types.issubset(
451                    {int, float, np.int64, np.float64, np.int32, np.float32, np.ndarray}
452                ):
453                    dataframe[col] = pd.to_numeric(dataframe[col], errors="coerce")
454                    dataframe[col] = dataframe[col].fillna(dataframe[col].mean(skipna=True))
455
456                    primitive_types = dataframe[col].dropna().map(type).unique().tolist()
457                    primitive_types = set(primitive_types)
458
459                    if len(primitive_types) > 1:
460                        raise ValueError(
461                            f"Mixed primitive types detected in column '{col}': {sorted(t.__name__ for t in primitive_types)}"
462                        )
463
464        return dataframe

466    def _fill_missing_values(self, df: pd.DataFrame) -> pd.DataFrame:
467        """Fill missing values in numeric columns with the mean of the column to ensure downstream processing can handle
468          numeric data without NaN values.
469
470        Args:
471            df: The input DataFrame to analyze and fill missing values in numeric columns.
472
473        Returns:
474            A DataFrame with missing values in numeric columns filled with the mean of the respective column,
475            while non-numeric columns are left unchanged.
476        """
477        dataframe = df.copy()
478
479        numeric_cols = dataframe.select_dtypes(include=[np.number]).columns
480
481        for col in numeric_cols:
482            mean_value = dataframe[col].mean(skipna=True)
483            if pd.notna(mean_value):
484                dataframe[col] = dataframe[col].fillna(mean_value)
485        return dataframe

487    def _detect_repeating_values(self, df: pd.DataFrame, threshold: int) -> tuple[bool, list[str]]:
488        """Identify columns that contain repeating temporal values, which may indicate time series data with regular intervals.
489        Args:
490            df: The input DataFrame to analyze for repeating temporal patterns.
491            threshold: An integer threshold for determining when a column has enough repeating values to be considered temporal
492
493        Returns:
494            A tuple containing a boolean indicating whether any repeating temporal columns were detected, and a list of
495
496            column names that exhibit repeating temporal patterns based on the specified threshold.
497
498
499        """
500        # detect repeating values for any mapping key value
501        dataframe = df.copy()
502        repeating_col: list[str] = []
503
504        for col in dataframe.columns:
505            value_counts = dataframe[col].value_counts()
506            if (
507                not value_counts.empty and value_counts.iloc[0] > threshold
508            ):  # arbitrary threshold for "repeating"
509                repeating_col.append(col)
510
511        return (bool(repeating_col), repeating_col)
